/*---------------------- Function Prototypes -------------------------*/
void UserScreenToDwg ();

void UserScreenToDwg (
    ProDrawing drawing,
    ProPoint3d scn_point,
    ProPoint3d drw_point)
{
    ProName    w_size;
    ProMatrix  matrix;
    int cur_sheet;
    ProError err;

    err = ProDrawingCurrentSheetGet(drawing, &cur_sheet);
    err = ProDrawingSheetTrfGet(drawing, cur_sheet, w_size, matrix);

    err = ProPntTrfEval(scn_point, matrix, drw_point);
}
