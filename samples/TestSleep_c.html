#include "TestSleep.h"


/*====================================================================*\
    FUNCTION :  ProTestSleep()
    PURPOSE  :  Machine specific Sleep in seconds. 
\*====================================================================*/
int ProTestSleep (int seconds)
{
#ifdef PRO_OS
#if (PRO_OS == 3) /* UNIX */
  sleep (seconds);
#elif(PRO_OS == 4)  /* WINDOWS */
  Sleep (seconds * 1000);
#endif
#endif

  return 0;
}
